1- create a directory

--- install pkg ---
1- create directory "src"
2- ros2 pkg create --build-type ament_cmake --node-name my_node my_package
3- colcon build  --packages-select my_package
4- source intsall/local_setup.bash
5- ros2 run my_package my_node
